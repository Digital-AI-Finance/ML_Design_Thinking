\begin{frame}{Validation \& Final Selection}
\begin{columns}[T]
\begin{column}{0.55\textwidth}
\small
\textbf{\stagecolor{refine}{Validation} -- Testing (500 $\rightarrow$ 50)}

\sigeq{\text{CV} = \frac{1}{k}\sum_{i=1}^{k}\text{Score}(f_{-i}, D_i)}

Cross-validation ensures generalization

\vspace{0.3em}
\textbf{\stagecolor{refine}{A/B Testing} -- Comparison}

\sigeq{t = \frac{\bar{x}_A - \bar{x}_B}{\sqrt{\frac{s_A^2}{n_A} + \frac{s_B^2}{n_B}}}}

Statistical significance for decisions

\vspace{0.3em}
\textbf{\stagecolor{strategy}{Responsible AI} -- Selection (50 $\rightarrow$ 5)}

\sigeq{\phi_j = \sum_{S}\frac{|S|!(n-|S|-1)!}{n!}[v(S \cup j) - v(S)]}

SHAP values for explainability
\end{column}

\begin{frame}{Key Takeaways \& Decision Framework}
\begin{columns}[T]
\begin{column}{0.48\textwidth}
\textbf{5 Key Insights:}
\begin{enumerate}
\item ML \textbf{amplifies} human innovation
\item \textbf{Both phases} are essential
\item \textbf{Match technique} to phase
\item \textbf{Watch for pitfalls} at each stage
\item \textbf{Trust but verify} with validation
\end{enumerate}

\vspace{0.5em}
\textbf{Common Pitfalls:}
\begin{itemize}
\item \textcolor{generate}{Divergent}: Analysis paralysis, noise over signal
\item \textcolor{strategy}{Convergent}: Premature closure, overfitting
\end{itemize}
\end{column}

\begin{column}{0.48\textwidth}
\textbf{Decision Framework:}
\begin{itemize}
\item Need to explore? $\rightarrow$ Unsupervised
\item Need to predict? $\rightarrow$ Supervised
\item Need to generate? $\rightarrow$ Generative AI
\item Need to group? $\rightarrow$ Clustering
\item Need to rank? $\rightarrow$ Classification
\item Need to test? $\rightarrow$ A/B Testing
\end{itemize}
